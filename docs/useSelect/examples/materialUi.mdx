# MaterialUI

A dropdown created with MaterialUI components such as `<FormLabel>`, `<List>`, `<ListItem>` and `<Button>`. Also child components
that are needed to display the `<ListItem>`.

https://material-ui.com/

### Playground

<Playground style={{height: '50px'}}>
  {() => {
    const DropdownSelectMaterialUI = () => {
      const {
        isOpen,
        selectedItem,
        getToggleButtonProps,
        getLabelProps,
        getMenuProps,
        highlightedIndex,
        getItemProps,
      } = useSelect({
        items: options,
        itemToString: item => item.primary,
      })
      const toggleButtonProps = getToggleButtonProps({
        refKey: 'toggleButtonRef',
      })
      const menuProps = getMenuProps({
        refKey: 'menuRef',
      })
      const {toggleButtonRef} = toggleButtonProps
      const {menuRef} = menuProps
      delete toggleButtonProps.toggleButtonRef
      delete menuProps.menuRef
      return (
        <>
          <FormLabel {...getLabelProps()}>My Cloud:</FormLabel>
          <Button
            variant="contained"
            color="secondary"
            className={classes.button}
            ref={toggleButtonRef}
            {...toggleButtonProps}
          >
            {selectedItem ? selectedItem.primary : 'Folders'}
            <ExpandMoreIcon className={classes.rightIcon} />
          </Button>
          <List className={classes.root} ref={menuRef} {...menuProps}>
            {isOpen && options.map((option, index) => {
              const itemProps = getItemProps({item: option, index, refKey: 'itemRef'})
              const {itemRef} = itemProps
              delete itemProps.itemRef
              return (
                <ListItem 
                  key={`${option.primary}-${index}`} 
                  className={index === highlightedIndex ? classes.highlighted : undefined} 
                  ref={itemRef} 
                  {...itemProps}
                >
                  <ListItemAvatar>
                    <Avatar>
                      {React.createElement(option.icon)}
                    </Avatar>
                  </ListItemAvatar>
                  <ListItemText primary={option.primary} secondary={option.secondary} />
                </ListItem>
              )
            })}
          </List>
        </>
      )
    }
    const useStyles = makeStyles(theme => ({
      root: {
        width: '100%',
        maxWidth: 300,
        backgroundColor: theme.palette.background.paper,
        maxHeight: 250,
        overflowY: 'auto',
        position: 'absolute',
        margin: 0,
        borderTop: 0,
      },
      highlighted: {
        backgroundColor: '#bde4ff',
      },
      button: {
        margin: theme.spacing(1),
      },
      rightIcon: {
        marginLeft: theme.spacing(1),
      },
    }))
    const classes = useStyles()
    const options = [
      {
        primary: 'Photos',
        secondary: 'Jan 9, 2014',
        icon: ImageIcon,
      },
      {
        primary: 'Work',
        secondary: 'Jan 7, 2014',
        icon: WorkIcon,
      },
      {
        primary: 'Vacation',
        secondary: 'July 20, 2014',
        icon: BeachAccessIcon,
      },
      {
        primary: 'Birthday',
        secondary: 'July 22, 2014',
        icon: Cake,
      },
      {
        primary: 'Friends',
        secondary: 'August 12, 2014',
        icon: Group,
      },
      {
        primary: 'New Car',
        secondary: 'September 1, 2014',
        icon: DirectionsCar,
      },
    ]
    return <DropdownSelectMaterialUI />
}}

</Playground>
